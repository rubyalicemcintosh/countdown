  <script>

    (function () {

      "use strict";


      const targetTime = new Date("2026-04-07T23:00:00Z").getTime();


      const daysEl = document.getElementById("days");

      const hoursEl = document.getElementById("hours");

      const minutesEl = document.getElementById("minutes");

      const secondsEl = document.getElementById("seconds");

      const countdownEl = document.getElementById("countdown");

      const subheadingEl = document.getElementById("subheading");

      const liveMessageEl = document.getElementById("live-message");


      function pad(value) {

        return String(value).padStart(2, "0");

      }


      function updateCountdown() {

        const now = Date.now();

        const diff = targetTime - now;


        if (diff <= 0) {

          daysEl.textContent = "00";

          hoursEl.textContent = "00";

          minutesEl.textContent = "00";

          secondsEl.textContent = "00";

          subheadingEl.textContent = "LAUNCHED";

          countdownEl.hidden = true;

          liveMessageEl.hidden = false;

          return;

        }


        const totalSeconds = Math.floor(diff / 1000);

        const days = Math.floor(totalSeconds / 86400);

        const hours = Math.floor((totalSeconds % 86400) / 3600);

        const minutes = Math.floor((totalSeconds % 3600) / 60);

        const seconds = totalSeconds % 60;


        daysEl.textContent = String(days);

        hoursEl.textContent = pad(hours);

        minutesEl.textContent = pad(minutes);

        secondsEl.textContent = pad(seconds);

      }


      updateCountdown();

      setInterval(updateCountdown, 1000);


      const canvas = document.getElementById("confetti-canvas");

      const ctx = canvas.getContext("2d", { alpha: true });

      const colors = ["#FCF2D1", "#FDCBE6", "#E8F7C8"];

      const particles = [];

      const prefersReducedMotion = window.matchMedia("(prefers-reduced-motion: reduce)").matches;


      let width = 0;

      let height = 0;


      function resizeCanvas() {

        const ratio = Math.max(window.devicePixelRatio || 1, 1);

        width = window.innerWidth;

        height = window.innerHeight;

        canvas.width = Math.floor(width * ratio);

        canvas.height = Math.floor(height * ratio);

        canvas.style.width = width + "px";

        canvas.style.height = height + "px";

        ctx.setTransform(ratio, 0, 0, ratio, 0, 0);

      }


      function createParticle() {

        return {

          x: Math.random() * width,

          y: -20 - Math.random() * height * 0.25,

          w: 6 + Math.random() * 8,

          h: 8 + Math.random() * 10,

          vx: -1.2 + Math.random() * 2.4,

          vy: 1.6 + Math.random() * 2.4,

          rotation: Math.random() * Math.PI * 2,

          vr: -0.08 + Math.random() * 0.16,

          color: colors[Math.floor(Math.random() * colors.length)]

        };

      }


      function seedParticles() {

        particles.length = 0;

        const count = Math.max(40, Math.floor(width / 18));


        for (let i = 0; i < count; i++) {

          const p = createParticle();

          p.y = Math.random() * height;

          particles.push(p);

        }

      }


      function step() {

        ctx.clearRect(0, 0, width, height);


        for (let i = 0; i < particles.length; i++) {

          const p = particles[i];

          p.x += p.vx;

          p.y += p.vy;

          p.rotation += p.vr;


          if (p.y > height + 24) {

            particles[i] = createParticle();

            particles[i].y = -20;

            continue;

          }


          if (p.x < -30) p.x = width + 30;

          if (p.x > width + 30) p.x = -30;


          ctx.save();

          ctx.translate(p.x, p.y);

          ctx.rotate(p.rotation);

          ctx.fillStyle = p.color;

          ctx.fillRect(-p.w / 2, -p.h / 2, p.w, p.h);

          ctx.restore();

        }


        window.requestAnimationFrame(step);

      }


      resizeCanvas();


      if (!prefersReducedMotion) {

        seedParticles();

        step();

      }


      window.addEventListener("resize", function () {

        resizeCanvas();

        seedParticles();

      });

    })();
